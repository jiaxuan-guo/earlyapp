CMAKE_MINIMUM_REQUIRED(VERSION 3.12)

# Project name and version
PROJECT("earlyapp")
SET(PROJECT_VERSION_MAJOR 1)
SET(PROJECT_VERSION_MINOR 0)
SET(PROJECT_VERSION_PATCH 20)

# Version file.
CONFIGURE_FILE("version.h.in" ${CMAKE_BINARY_DIR}/include/EAVersion.h)

# Program executable.
SET(PROGRAM_EXE ${CMAKE_PROJECT_NAME})


# [Features]
#  - Log output
OPTION(USE_LOGOUTPUT "Control detailed log output to standard out" OFF)
IF(USE_LOGOUTPUT)
    ADD_DEFINITIONS(-DUSE_LOGOUTPUT)
ENDIF(USE_LOGOUTPUT)

#  - Log demsg output
OPTION(USE_DMESGLOG "Log output to dmesg log" OFF)
IF(USE_DMESGLOG)
    ADD_DEFINITIONS(-DUSE_DMESGLOG)
ENDIF(USE_DMESGLOG)


SUBDIRS(ext/MediaSDK/src ext/CameraICI/src ext/CameraCSI/src ext/GLES2/src src)

# Service configuration
SUBDIRS(config)

# Resources files
SUBDIRS(res)

SUBDIRS(fastboot)
